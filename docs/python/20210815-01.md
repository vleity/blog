```python
import requests
import datetime
import json
import urllib3
urllib3.disable_warnings()


class WorkWeChat:
    def __init__(self, corp_id):
        """
        初始化企业
        """
        self.corp_id = corp_id

    class WorkApp:
        def __init__(self, obj, agent_id, secret):
            """
            初始化企业应用
            """
            self.corp_id = obj.corp_id
            self.agent_id = agent_id
            self.secret = secret

        def get_token(self):
            """
            获取token
            """
            url = "https://qyapi.weixin.qq.com/cgi-bin/gettoken?corpid={}&corpsecret={}".format(self.corp_id, self.secret)
            r = requests.get(url=url)
            token = r.json()['access_token']
            return token
        
        def send_text_message(self, userid, message):
            """
            发送文本消息
            :param userid: 指定接收消息的成员，成员ID列表（多个接收者用‘|’分隔，最多支持1000个）。特殊情况：指定为”@all”，则向该企业应用的全部成员发送
            :param message: 文字内容
            """
            token = self.get_token()
            url = "https://qyapi.weixin.qq.com/cgi-bin/message/send?access_token={}".format(token)
            data = {
                "touser": userid,
                "msgtype": "text",
                "agentid": self.agent_id,
                "text": {
                    "content": message
                },
                "safe": 0,
                "enable_id_trans": 0,
                "enable_duplicate_check": 0,
                "duplicate_check_interval": 1800
            }
            resp = requests.post(url=url, data=json.dumps(data), verify=False).json()
            log_time = datetime.datetime.now().strftime('%Y-%m-%d %H:%M:%S')
            if resp['errcode'] == 0 and resp.get('invaliduser','') == '':
                print('{} - INFO - {}'.format(log_time,'企业微信发送成功'))
            elif resp['errcode'] == 0 and resp.get('invaliduser','') != '':
                print('{} - WARNING - {}'.format(log_time,'企业微信发送成功，%s无效'% resp['invaliduser']))
            else:
                print('{} - FAIL - {}'.format(log_time,'企业微信发送失败，%s失败'% resp['invaliduser']))
            
            
        def send_markdown_message(self, userid, content):
            """
            发送文本消息
            :param userid: 指定接收消息的成员，成员ID列表（多个接收者用‘|’分隔，最多支持1000个）。特殊情况：指定为”@all”，则向该企业应用的全部成员发送
            :param userid: markdown内容
            """
            token = self.get_token()
            url = "https://qyapi.weixin.qq.com/cgi-bin/message/send?access_token={}".format(token)
            data = {
                   "touser" : userid,
                   "msgtype": "markdown",
                   "agentid" : self.agent_id,
                   "markdown": {
                       "content": content 
                   },
                   "enable_duplicate_check": 0,
                   "duplicate_check_interval": 1800
                }

            resp = requests.post(url=url, data=json.dumps(data), verify=False).json()
            log_time = datetime.datetime.now().strftime('%Y-%m-%d %H:%M:%S')
            if resp['errcode'] == 0 and resp.get('invaliduser','') == '':
                print('{} - INFO - {}'.format(log_time,'企业微信发送成功'))
            elif resp['errcode'] == 0 and resp.get('invaliduser','') != '':
                print('{} - WARNING - {}'.format(log_time,'企业微信发送成功，%s无效'% resp['invaliduser']))
            else:
                print('{} - FAIL - {}'.format(log_time,'企业微信发送失败，%s失败'% resp['invaliduser']))

                
        def send_image_message(self, userid, path=None, bio=None):
            """
            发送文本消息
            :param userid: 指定接收消息的成员，成员ID列表（多个接收者用‘|’分隔，最多支持1000个）。特殊情况：指定为”@all”，则向该企业应用的全部成员发送
            :param userid: 图片
            """
            token = self.get_token()
            # 上传临时素材并获取 media_id
            upload_url = "https://qyapi.weixin.qq.com/cgi-bin/media/upload?access_token={}&type=image".format(token)
            if path:
                upload_data = {'media': open(path, 'rb')}
            if bio:
                upload_data = {'media': bio}
            upload_r = requests.post(upload_url, files=upload_data, verify=False).json()
            log_time = datetime.datetime.now().strftime('%Y-%m-%d %H:%M:%S')
            if upload_r.get('errcode',-1) == 0:
                print('{} - INFO - {}'.format(log_time,'上传临时素材成功'))
                media_id = upload_r.get('media_id','-1')
                # 发送图片
                url = "https://qyapi.weixin.qq.com/cgi-bin/message/send?access_token={}".format(token)
                data = {
                       "touser" : userid,
                       "msgtype": "image",
                       "agentid" : self.agent_id,
                       "image": {
                           "media_id": media_id 
                       },
                       "enable_duplicate_check": 0,
                       "duplicate_check_interval": 1800
                    }

                resp = requests.post(url=url, data=json.dumps(data), verify=False).json()
                if resp['errcode'] == 0 and resp.get('invaliduser','') == '':
                    print('{} - INFO - {}'.format(log_time,'企业微信发送成功'))
                elif resp['errcode'] == 0 and resp.get('invaliduser','') != '':
                    print('{} - WARNING - {}'.format(log_time,'企业微信发送成功，%s无效'% resp['invaliduser']))
                else:
                    print('{} - FAIL - {}'.format(log_time,'企业微信发送失败，%s失败'% resp['invaliduser']))
            else:
                print('{} - WARNING - {}'.format(log_time,'上传临时素材失败，%s'% upload_r['errmsg']))


# 创建企业
corp = WorkWeChat(corp_id = 'xxx')

# 创建企业应用
app1 = corp.WorkApp(corp, agent_id = 'xxx', secret = 'xxxxxx')
# 发送应用消息
app1.send_text_message('LeiTaiYang','hello')


# 创建企业应用
app2 = corp.WorkApp(corp, agent_id = 'xxx', secret = 'xxxxxx')
# 发送应用消息
app2.send_text_message('@all','Ptrade','test')

```
